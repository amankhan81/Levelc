        const messageBox = document.getElementById('messageBox');
        let messageTimeout;
        
        /**
         * Displays a temporary status or error message.
         */
        function showMessage(message, isError = false) {
            clearTimeout(messageTimeout);
            messageBox.textContent = message;
            messageBox.classList.remove('hidden', 'opacity-0');
            messageBox.classList.add('opacity-100');

            if (isError) {
                messageBox.className = 'fixed bottom-4 right-4 p-4 text-sm font-medium rounded-lg shadow-lg transition-opacity duration-300 z-50 bg-red-100 text-red-700';
            } else {
                // If it's a success message, we now rely on the caller not calling showMessage on success.
                messageBox.className = 'fixed bottom-4 right-4 p-4 text-sm font-medium rounded-lg shadow-lg transition-opacity duration-300 z-50 bg-green-100 text-green-700';
            }

            messageTimeout = setTimeout(() => {
                messageBox.classList.remove('opacity-100');
                messageBox.classList.add('opacity-0');
                setTimeout(() => messageBox.classList.add('hidden'), 300);
            }, 5000);
        }

        // Global variable to track the currently playing audio
        let activeAudio = null; 

        /**
         * Stops any active audio and plays the new file based on the word.
         * @param {string} word The vocabulary word (used as the base filename).
         */
        function playAudio(word) {
            // Generates the filename like 'accord.mp3'
            const filename = `${word}.mp3`;

            // 1. Stop any currently playing audio
            if (activeAudio) {
                activeAudio.pause();
                activeAudio.currentTime = 0; // Rewind for next time
                activeAudio = null;
            }

            // 2. Create the new audio object and set it as active
            const audio = new Audio(filename);
            activeAudio = audio;

            // 3. Play the audio
            audio.play()
                .then(() => {
                    // Success: Message is intentionally suppressed here (per prior request).
                })
                .catch(error => {
                    console.error("Audio playback failed:", error);
                    activeAudio = null;
                    let errorMessage = `Failed to play ${filename}. Ensure the audio file is available in the same directory.`;
                    
                    if (error.name === 'NotAllowedError') {
                        errorMessage = "Playback blocked by browser. Try interacting with the page first.";
                    }
                    
                    showMessage(errorMessage, true);
                });

            // 4. Clear activeAudio when the current one finishes
            audio.onended = () => {
                if (activeAudio === audio) {
                    activeAudio = null;
                }
            };
        }
